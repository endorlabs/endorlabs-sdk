# Resource Relationships

> **Understanding relationships between Endor Labs platform resources**

## 🔗 **Relationship Overview**

### **Hierarchical Structure**
```
Tenant
├── Namespace (tenant.namespace)
│   ├── Project (repository)
│   │   ├── Finding (vulnerability)
│   │   ├── Scan (analysis run)
│   │   └── Policy (security rule)
│   └── Policy (namespace-level)
└── Namespace (tenant.other-namespace)
    └── Project (other repository)
```

### **Resource Dependencies**
- **Namespaces** contain Projects, Policies, and other resources
- **Projects** generate Findings through Scans
- **Policies** apply to Projects and Namespaces
- **Scans** analyze Projects and produce Findings

## 📁 **Namespace Relationships**

### **Parent-Child Namespaces**
```python
# Parent namespace
parent_namespace = "tenant.namespace"

# Child namespace
child_namespace = f"{parent_namespace}.child"

# Grandchild namespace
grandchild_namespace = f"{child_namespace}.grandchild"
```

### **Namespace Containment**
```python
# Namespace contains projects
projects = list_projects(client, namespace_uuid)

# Namespace contains policies
policies = list_policies(client, namespace_uuid)

# Namespace contains other namespaces
child_namespaces = list_namespaces(client, parent_namespace)
```

### **Permission Inheritance**
```python
# Child namespaces inherit parent permissions
def check_namespace_permissions(client, namespace_uuid):
    """Check permissions for namespace and all children."""
    permissions = get_namespace_permissions(client, namespace_uuid)
    
    # Check child namespaces
    children = list_namespaces(client, namespace_uuid)
    for child in children:
        child_permissions = check_namespace_permissions(client, child.uuid)
        permissions.merge(child_permissions)
    
    return permissions
```

## 🏗️ **Project Relationships**

### **Project-Namespace Relationship**
```python
# Project belongs to namespace
project = get_project(client, project_uuid)
namespace = get_namespace(client, project.namespace_uuid)

# Namespace contains projects
projects = list_projects(client, namespace.uuid)
```

### **Project-Finding Relationship**
```python
# Project generates findings
findings = list_findings(client, project_uuid)

# Finding belongs to project
finding = get_finding(client, finding_uuid)
project = get_project(client, finding.project_uuid)
```

### **Project-Scan Relationship**
```python
# Project is analyzed by scans
scans = list_scans(client, project_uuid)

# Scan analyzes project
scan = get_scan(client, scan_uuid)
project = get_project(client, scan.project_uuid)
```

### **Project-Policy Relationship**
```python
# Policy applies to project
policies = list_policies_for_project(client, project_uuid)

# Project is subject to policies
def apply_policies_to_project(client, project_uuid):
    """Apply all applicable policies to project."""
    # Get namespace policies
    project = get_project(client, project_uuid)
    namespace_policies = list_policies(client, project.namespace_uuid)
    
    # Get project-specific policies
    project_policies = list_policies_for_project(client, project_uuid)
    
    # Combine and apply policies
    all_policies = namespace_policies + project_policies
    return apply_policies(all_policies, project)
```


### **Finding-Scan Relationship**
```python
# Finding is generated by scan

scan = get_scan(client, finding.scan_uuid)

# Scan produces findings

```

### **Finding-Policy Relationship**
```python
# Finding violates policy
policy_violations = list_policy_violations(client, finding_uuid)

# Policy detects findings
def check_policy_violations(client, finding_uuid):
    """Check if finding violates any policies."""
    finding = get_finding(client, finding_uuid)
    project = get_project(client, finding.project_uuid)
    
    # Get applicable policies
    policies = list_policies_for_project(client, project.uuid)
    
    # Check for violations
    violations = []
    for policy in policies:
        if violates_policy(finding, policy):
            violations.append(policy)
    
    return violations
```

### **Policy-Project Relationship**
```python

policies = list_policies_for_project(client, project_uuid)

def apply_project_policies(client, project_uuid):
    """Apply all applicable policies to project."""
    project = get_project(client, project_uuid)
    
    # Get namespace policies
```

### **Policy-Finding Relationship**
```python
# Policy detects findings
findings = list_findings_by_policy(client, policy_uuid)

# Finding violates policy
def check_finding_policy_violations(client, finding_uuid):
    """Check if finding violates any policies."""
    finding = get_finding(client, finding_uuid)
    project = get_project(client, finding.project_uuid)
    
    # Get applicable policies
    policies = list_policies_for_project(client, project.uuid)
```

```python
    """Run scan with policy enforcement."""
    # Get applicable policies
    policies = list_policies_for_project(client, project_uuid)
    
    # Run scan with policy enforcement
    scan = create_scan(client, project_uuid, {
        "type": "POLICY_SCAN",
        "policies": [p.uuid for p in policies]
```

## 🧪 **Relationship Testing**

### **Namespace Hierarchy Testing**
```python
def test_namespace_hierarchy(api_client, tenant_namespace):
    """Test namespace hierarchy relationships."""
    # Create parent namespace
    parent = create_namespace(api_client, tenant_namespace, {
        "name": "parent",
        "description": "Parent namespace"
    })
    
    # Create child namespace
    child = create_namespace(api_client, f"{tenant_namespace}.parent", {
        "name": "child",
        "description": "Child namespace"
    })
    
    # Test hierarchy
    assert child.meta.name == "child"
    assert parent.meta.name == "parent"
    
    # Test containment
    children = list_namespaces(api_client, f"{tenant_namespace}.parent")
    assert len(children) > 0
```

### **Policy Relationships Testing**
```python
def test_policy_relationships(api_client, namespace_uuid):
    """Test policy relationships."""
    # Create policy
    policy = create_policy(api_client, namespace_uuid, {
        "name": "test-policy",
        "description": "Test policy",
        "type": "SECURITY"
    })
    
    # Test namespace relationship

    # Test policy application
    projects = list_projects(api_client, namespace_uuid)
    for project in projects:

        assert policy.uuid in [p.uuid for p in applicable_policies]
```
